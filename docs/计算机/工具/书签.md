def add_pdf_bookmark():
    # 创建PDF写入对象，清空原有书签
    writer = PdfWriter()
    with open(INPUT_PDF, "rb") as src_file:
        writer.append(src_file, import_outline=False)

    # 记录各级父书签节点
    parent_chapter = None   # 一级父节点（章）
    parent_section = None   # 二级父节点（节）

    for title, book_page, level in catalog:
        # 转换为pypdf所需的0起始页面索引
        pdf_page_index = book_page + OFFSET - 1

        if level == 1:
            # 一级章：父级节点，设置默认折叠 open=False
            item = writer.add_outline_item(
                title, pdf_page_index, parent=None, is_open=True
            )
            parent_chapter = item
            parent_section = None  # 切换章节时重置二级父节点

        elif level == 2:
            # 二级节：父级节点，设置默认折叠 open=False
            item = writer.add_outline_item(
                title, pdf_page_index, parent=parent_chapter, is_open=True
            )
            parent_section = item

        elif level == 3:
            # 三级小节：末级节点，无子书签，无需设置open
            writer.add_outline_item(
                title, pdf_page_index, parent=parent_section, is_open=False
            )

    # 输出最终PDF文件
    with open(OUTPUT_PDF, "wb") as out_file:
        writer.write(out_file)

    print(f"✅ 全部父级书签默认折叠设置完成！")
    print(f"📄 输出文件：{OUTPUT_PDF}")


if __name__ == "__main__":
    add_pdf_bookmark()
